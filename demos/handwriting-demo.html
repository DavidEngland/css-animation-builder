Usage Examples

JavaScript Usage

// Basic handwriting effect
const handwriting = new HandwritingAnimation('#my-element', {
    speed: 50,
    penType: 'fountain',
    strokeColor: 'var(--text-primary)',
    strokeWidth: 2,
    showPen: true
});

handwriting.writeText('Hello, World!');

// Advanced signature animation
const signature = new HandwritingAnimation('#signature-element', {
    speed: 30,
    penType: 'quill',
    strokeColor: '#1a1a1a',
    strokeWidth: 3,
    showPen: true,
    penSize: 25,
    inkFlow: true,
    pressure: 1.5,
    shakiness: 0.2,
    style: 'signature',
    onComplete: () => console.log('Signature complete!')
});

signature.writeText('David England');

// Method chaining
handwriting
    .writeText('First line')
    .pause()
    .resume()
    .reset();

CSS Configuration

/* Handwriting container */
.handwriting-container {
    position: relative;
    font-family: 'Dancing Script', cursive;
    font-size: var(--font-size-xxl);
    color: var(--text-primary);
}

/* Custom pen styles */
.handwriting-path {
    stroke: #1a1a1a;
    stroke-width: 2;
    stroke-linecap: round;
    stroke-linejoin: round;
    fill: none;
}

/* Paper effects */
.lined-paper {
    background-image: repeating-linear-gradient(
        transparent,
        transparent 1.8em,
        var(--text-light) 1.8em,
        var(--text-light) calc(1.8em + var(--border-width-thin))
    );
}

/* Ink effects */
.ink-flowing path {
    filter: drop-shadow(0 0 var(--border-width-medium) rgba(0, 0, 0, 0.3));
}
