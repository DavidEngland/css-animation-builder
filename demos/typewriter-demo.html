Usage Examples

JavaScript Usage

// Basic typewriter effect
const typewriter = new TypewriterAnimation('#my-element', {
    speed: 100,
    cursor: '|',
    cursorBlink: true,
    humanize: true
});

typewriter.type('Hello, World!');

// Multiple sequences
typewriter.type([
    'Welcome to our website!',
    'We create amazing animations.',
    'Try our CSS Animation Builder!'
]);

// Advanced options
const advancedTypewriter = new TypewriterAnimation('#advanced-element', {
    speed: 80,
    deleteSpeed: 40,
    pauseTime: 2000,
    loop: true,
    deleteAll: true,
    humanize: true,
    mistakes: true,
    mistakeChance: 0.1,
    onComplete: () => console.log('Animation complete!')
});

// Method chaining
typewriter
    .type('First message')
    .pause()
    .resume()
    .reset();

CSS-Only Alternative

/* CSS-only typewriter effect */
.typewriter-text {
    overflow: hidden;
    border-right: var(--border-width-medium) solid;
    white-space: nowrap;
    margin: 0 auto;
    letter-spacing: 0.1em;
    animation: 
        typing 3s steps(40, end),
        blink-caret 0.75s step-end infinite;
}

@keyframes typing {
    from { width: 0 }
    to { width: var(--element-width-full) }
}

@keyframes blink-caret {
    from, to { border-color: transparent }
    50% { border-color: currentColor }
}
