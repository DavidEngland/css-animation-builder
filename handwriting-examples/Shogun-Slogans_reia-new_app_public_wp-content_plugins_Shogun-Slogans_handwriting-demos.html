// Typewriter functionality class SimpleTypewriter { constructor(element) { this.element = element; this.textElement = element.querySelector('.typewriter-text'); this.cursorElement = element.querySelector('.typewriter-cursor'); this.settings = { text: element.dataset.text || 'Sample text...', speed: parseInt(element.dataset.speed) || 100, cursor: element.dataset.cursor || '|', loop: element.dataset.loop !== 'false' }; this.currentIndex = 0; this.isTyping = false; this.timeoutId = null; this.init(); } init() { this.textElement.textContent = ''; this.cursorElement.textContent = this.settings.cursor; this.startTyping(); } startTyping() { if (this.isTyping) return; this.isTyping = true; this.typeCharacter(); } typeCharacter() { const currentText = this.settings.text; const displayText = currentText.substring(0, this.currentIndex + 1); this.textElement.textContent = displayText; this.currentIndex++; if (this.currentIndex < currentText.length) { this.timeoutId = setTimeout(() => this.typeCharacter(), this.settings.speed); } else { this.isTyping = false; if (this.settings.loop) { this.timeoutId = setTimeout(() => { this.currentIndex = 0; this.textElement.textContent = ''; this.startTyping(); }, 2000); } } } } // Initialize all typewriters document.addEventListener('DOMContentLoaded', function() { const typewriters = document.querySelectorAll('.shogun-typewriter'); typewriters.forEach(element => { new SimpleTypewriter(element); }); });
